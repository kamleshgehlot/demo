CREATE TABLE `test`.`user` ( `id` INT NOT NULL AUTO_INCREMENT , `first_name` VARCHAR(100) NULL DEFAULT NULL , `last_name` VARCHAR(100) NULL DEFAULT NULL , `email` VARCHAR(100) NULL DEFAULT NULL , `country_code` VARCHAR(20) NULL DEFAULT NULL , `mobile` VARCHAR(15) NULL DEFAULT NULL , `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`id`)) ENGINE = InnoDB;
